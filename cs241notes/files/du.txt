du      // estimate file space usage

-a      // write counts for all files, not just directories

-c      // produce a grand total

-h      // print sizes in human readable format (e.g., 1K 234M 2G)

-S      // for directories do not include size of subdirectories

-s      // display only a total for each argument

-t      // exclude entries smaller than SIZE if positive, or entries greater
            than SIZE if negative

--time      // show  time of the last modification of any file in the
                directory, or any of its subdirectories

--exclude=PATTERN   // exclude files that match PATTERN

-x      // skip directories on different file systems

-b or --bytes   // Prints size of directory in bytes

-m      // Prints size of directory in MB
